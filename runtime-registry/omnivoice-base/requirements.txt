# /v1/generate, /v1/variants/build, /v1/metadata over HTTP/JSON.
fastapi==0.115.0
uvicorn[standard]==0.32.0
pydantic==2.9.2

# The OmniVoice TTS framework.
# Diffusion-language-model TTS finetuned from Qwen3-0.6B-Base.
# 0.6B params, ~646 languages, voice cloning + voice design +
# emotion tags. CPU-capable (float32, slow). GPU recommended for
# production (RTF 0.025 on H100 vendor-reported).
#
# The runtime entrypoint loads the model on the first /v1/generate
# request (lazy), guarded by a lock so concurrent first requests
# share the same load. If the framework is unavailable in the
# deployment environment, the runtime reports "load_failed" on
# /ready and the manager reaps it on idle_timeout.
omnivoice==0.1.5
